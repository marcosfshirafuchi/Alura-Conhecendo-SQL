SELECT strftime('%Y',data_venda) from vendas LIMIT 5;

SELECT DISTINCT (strftime('%Y',data_venda)) from vendas;

SELECT DISTINCT (strftime('%Y',data_venda)) as Ano from vendas
ORDER By Ano;

SELECT DISTINCT (strftime('%Y',data_venda)) AS Ano, COUNT(id_venda) AS Total_Vendas
from vendas
GROUP BY Ano
ORDER By Ano;

SELECT DISTINCT (strftime('%Y',data_venda)) AS Ano, strftime('%m',data_venda) as Mes, COUNT(id_venda) AS Total_Vendas
from vendas
GROUP BY Ano, Mes
ORDER By Ano;
