SELECT DISTINCT (strftime('%Y',data_venda)) AS Ano, strftime('%m',data_venda) as Mes, COUNT(id_venda) AS Total_Vendas
from vendas
WHERE Mes = '01' or Mes = '11' OR Mes = '12'
GROUP BY Ano, Mes
ORDER By Ano;
